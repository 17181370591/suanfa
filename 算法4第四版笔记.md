#扑克牌排序
import random,numpy as np,time
l=np.arange(52)+1
np.random.shuffle(l)
print(l)

def ex(l,a,b):      #交换两张牌
    l[a],l[b]=l[b],l[a]

def f(l):           #希尔排序换牌
    h,le=1,len(l)
    while h<le/3:
        h=h*3+1
    print(h)
    while h>0:
        for i in range(h,le):
            for j in range(i,h-1,-h):
                if l[j]<l[j-h]:
                    l[j],l[j-h]=l[j-h],l[j]
        h=int(h/3)
    print(l)

f(l)
